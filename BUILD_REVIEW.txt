=== .github/workflows/build.yml ===
name: Build Flutter APK

on:
  push:
    branches:
      - main
      - fix/app-improvements
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Java
        uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: '17'

      - name: Set up Flutter
        uses: subosito/flutter-action@v2
        with:
          channel: stable

      - name: Generate official Flutter Android scaffold
        run: |
          rm -rf /tmp/hs_scaffold
          flutter create \
            --platforms=android \
            --project-name hamadasiana \
            --org uk.websocket.hamadasiana \
            /tmp/hs_scaffold

          rm -rf ./android
          cp -R /tmp/hs_scaffold/android ./android
          cp /tmp/hs_scaffold/.metadata ./.metadata

      - name: Add Android permissions and app label to AndroidManifest
        run: |
          python3 - <<'PY'
          from pathlib import Path
          p = Path("android/app/src/main/AndroidManifest.xml")
          s = p.read_text(encoding="utf-8")
          perms = [
              '<uses-permission android:name="android.permission.INTERNET"/>',
              '<uses-permission android:name="android.permission.ACCESS_FINE_LOCATION"/>',
              '<uses-permission android:name="android.permission.ACCESS_COARSE_LOCATION"/>',
              '<uses-permission android:name="android.permission.RECORD_AUDIO"/>',
              '<uses-permission android:name="android.permission.READ_CONTACTS"/>'
          ]
          if '<manifest xmlns:android="http://schemas.android.com/apk/res/android">' in s:
              for perm in perms:
                  if perm not in s:
                      s = s.replace(
                          '<manifest xmlns:android="http://schemas.android.com/apk/res/android">',
                          '<manifest xmlns:android="http://schemas.android.com/apk/res/android">\n    ' + perm,
                          1
                      )

          s = s.replace('android:label="hamadasiana"', 'android:label="Hamada Siana"')
          s = s.replace('android:label="hs_scaffold"', 'android:label="Hamada Siana"')

          p.write_text(s, encoding="utf-8")
          print(p.read_text(encoding="utf-8"))
          PY

      - name: Flutter pub get
        run: flutter pub get

      - name: Generate launcher icons
        run: flutter pub run flutter_launcher_icons

      - name: Build APK
        run: |
          flutter build apk --release
          mv build/app/outputs/flutter-apk/app-release.apk build/app/outputs/flutter-apk/Hamada-Siana.apk

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: Hamada Siana
          path: build/app/outputs/flutter-apk/Hamada-Siana.apk
